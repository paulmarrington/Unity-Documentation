30  /// <inheritdoc />
31  protected override void Initialise() {
32  // only use services that are valid for the current context
33  var useful = services.Where(service => service.context.Equals(context) && service.IsExternalServiceAvailable());
34  services = useful.ToArray();
35  // service processing order is dependent on the priority each gives
36  Array.Sort(array: services, comparison: (x, y) => x.priority.CompareTo(value: y.priority));
37 
38  selector = new Selector<TS> {
39  IsRandom = order > Order.RoundRobin
40  , ExhaustiveBelow = order == Order.RandomExhaustive ? services.Length + 1 : 0
41  , Choices = services
42  };
43  usagesRemaining = 0;
44  }

46  /// <a href="">Get the next service instance given selection order and repetitions</a> //#TBD#//
47  public TI Instance<TI>() where TI : TS {
48  if (order == Order.TopDown) {
49  selector.Top();
50  } else if (--usagesRemaining > 0) {
51  return (TI) currentService;
52  }
53  currentService = selector.Pick();
54  usagesRemaining = currentService.usageBalance;
55  return (TI) currentService;
56  }
57 
58  /// <a href="">If the last service fails, ask for another. If none work, returns null</a> //#TBD#//
59  public TI Next<TI>() where TI : TS {
60  currentService = selector.Next();
61  if (currentService == default) return default;
62  usagesRemaining = currentService.usageBalance;
63  return (TI) currentService;
64  }
