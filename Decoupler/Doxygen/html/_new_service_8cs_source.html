18  [MenuItem("Assets/Create/Decoupled/New Service")] private static void Phase1() {
19  var templatePath = TemplatePath();
20  var sources = AssetDatabase.FindAssets("", new[] {templatePath});
21  var destinationPath = EditorUtility.SaveFilePanel("Save Your New Service", GetSelectedPathOrFallback(), "", "");
22  var serviceName = Path.GetFileNameWithoutExtension(destinationPath);
23  if (string.IsNullOrEmpty(destinationPath)) return;
24  if (Directory.Exists(destinationPath)) {
25  Debug.LogError($"{destinationPath} already exists. Please select a different name or project directory");
26  return;
27  }
28  Directory.CreateDirectory(destinationPath);
29 
30  using (var template = Template.Instance) {
31  template.Substitute("Template", serviceName);
32  for (int i = 0; i < sources.Length; i++) {
33  var sourcePath = AssetDatabase.GUIDToAssetPath(sources[i]);
34  if (!File.Exists(sourcePath)) continue;
35  var text = template.Process(File.ReadAllText(sourcePath));
36  var fileName = Path.GetFileNameWithoutExtension(sourcePath);
37  File.WriteAllText($"{destinationPath}/{serviceName}{fileName}.cs", text);
38  }
39 
40  destinationPath = destinationPath.Substring(destinationPath.IndexOf("/Assets/", StringComparison.Ordinal) + 1);
41  AssetDatabase.ImportAsset(
42  Path.GetDirectoryName(destinationPath),
43  ImportAssetOptions.ForceSynchronousImport | ImportAssetOptions.ImportRecursive);
44  PlayerPrefs.SetString("DecouplerBuildServiceName", serviceName);
45  PlayerPrefs.SetString("DecouplerBuildDestinationPath", destinationPath);
46 
47  Debug.Log($"Scripts for `{destinationPath}` waiting on rebuild for basic assets...");
48  // Will continue in `OnScriptsReloaded`
49  }
50  }

59  Debug.Log($"... rebuild complete, creating basic assets for `{destinationPath}`");
60  Debug.Log($" 1. Update `{serviceName}Context.cs` with context data for your service. Don't forget `Equals`");
61  Debug.Log(
62  $" 2. Fill `{serviceName}ServiceAdapter.cs` with supporting code and abstract or default interface methods");
63  Debug.Log($" 3. Fill `{serviceName}ServiceForMock.cs` interface methods for mocking");
64  Debug.Log(" 4. Write some tests to prove that the mock interface works as expected");
65  Debug.Log($" 5. Write `{serviceName}ServiceForSource.cs` where `Source` is a service source");

71  Environment mockEnvironment =

91  servicesManagerSerializedObject.ApplyModifiedProperties();
92  contextSerializedObject.ApplyModifiedProperties();
93  serviceForMockSerializedObject.ApplyModifiedProperties();
94  AssetDatabase.SaveAssets();

106  private static void SetField(SerializedObject asset, string fieldName, Object fieldValue) {
107  var serialisedProperty = FindProperty(asset, fieldName);
108  if (serialisedProperty != null) serialisedProperty.objectReferenceValue = fieldValue;
109  }

111  private static void InsertIntoArrayField(SerializedObject asset, string fieldName, Object fieldValue) {
112  var serialisedProperty = FindProperty(asset, fieldName);
113  if (serialisedProperty != null) {
114  serialisedProperty.InsertArrayElementAtIndex(0);
115  var arrayElementSerialisedProperty = serialisedProperty.GetArrayElementAtIndex(0);
116  arrayElementSerialisedProperty.objectReferenceValue = fieldValue;
117  asset.ApplyModifiedProperties();
118  }
119  }

130  private static string TemplatePath() {
131  var paths = AssetDatabase.FindAssets("DecouplerTemplatePath");
132  for (int i = 0; i < paths.Length; i++) {
133  var path = AssetDatabase.GUIDToAssetPath(paths[i]);
134  if (path.IndexOf("Askowl", StringComparison.Ordinal) == -1) return path;
135  }
136  if (paths.Length == 0) return "";
137  return AssetDatabase.LoadAssetAtPath<String>(AssetDatabase.GUIDToAssetPath(paths[0]));
138  }

140  private static string GetSelectedPathOrFallback() {
141  string path = "Assets";
142  foreach (Object obj in Selection.GetFiltered(typeof(Object), SelectionMode.Assets)) {
143  path = AssetDatabase.GetAssetPath(obj);
144  if (!string.IsNullOrEmpty(path) && File.Exists(path)) {
145  path = Path.GetDirectoryName(path);
146  break;
147  }
148  }
149  return path;
