18  [SerializeField] private bool enabled = false;
19  [SerializeField] private float secondsDelayAtStart = 5;
20  [SerializeField] private float updateIntervalInSeconds = 1;
21 
22  /// <a href="http://bit.ly/2QR9q42"><see cref="OfType{T}.Initialise"/></a>
23  public void Initialise(WithEmitter componentToPoll) {
24  if (!enabled) return;
25 
26  void poll(Fiber fiber) {
27  if (!componentToPoll.Equals(null)) componentToPoll.Emitter.Fire();
28  }
29 
30  Fiber.Start.WaitFor(secondsDelayAtStart).Begin.Do(poll).WaitFor(updateIntervalInSeconds).Again.Finish();
31  }
32  }

154  private static void UnloadResources(PlayModeStateChange playModeState) {
155  if (playModeState != PlayModeStateChange.ExitingPlayMode) return;
156  EditorApplication.playModeStateChanged -= UnloadResources;
157  for (int i = assetsToUnload.Count - 1; i >= 0; i--) Resources.UnloadAsset(assetsToUnload[i]);
