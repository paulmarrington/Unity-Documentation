7 namespace CustomAsset.Mutable {
8  /// <a href="http://bit.ly/2CwSUh0">Change a float value in linear fashion over time</a>
9  [CreateAssetMenu(menuName = "Custom Assets/Mutable/Change Over Time")]
10  public class ChangeOverTime : Trigger {
11  [SerializeField] private Float targetForChange = default;
12  [SerializeField] private float amountToChange = 0;
13  [SerializeField] private float overSeconds = 1;
14  [SerializeField] private int stepsPerSecond = 10;
15  [SerializeField] private bool resetOnStart = false;
16 
17  private Fiber change;
18 
19  /// <a href="http://bit.ly/2CwSUh0">Start a fiber to change over non-default period</a>
20  public void Fire(float changeAmount, float seconds) {
21  base.Fire();
22  if (resetOnStart) targetForChange.Value = targetForChange.Minimum;
23  var steps = Math.Max((int) (stepsPerSecond * seconds), 1);
24  var stepTime = seconds / steps;
25  var stepAmount = changeAmount / steps;
26 
27  //- we heal in steps, not quitting on full health because player may also be taking damage
28  void step(Fiber fiber) => targetForChange.Set(targetForChange + stepAmount);
29 
30  void finish(Fiber fiber) => change = null;
31 
32  //- Fibers do not need to run from a MonoBehaviour
33  change = Fiber.Start.Begin.Do(step).WaitFor(stepTime).Repeat(steps).Do(finish);
34  }
35 
36  /// <a href="http://bit.ly/2CwSUh0">Start a fiber to change over default period</a> <inheritdoc />
37  public override void Fire() => Fire(amountToChange, overSeconds);
38 
39  /// <a href="http://bit.ly/2CwSUh0">Abort a long-running change loop</a>
40  public void Abort() => change?.Exit();
41  }
42 }
