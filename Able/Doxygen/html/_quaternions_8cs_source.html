18  public static float LengthSquared(this Quaternion q) => q.x * q.x + q.y * q.y + q.z * q.z + q.w * q.w;

20  /// <a href="http://bit.ly/2NUH4V6"> A faster normalise, accurate to a dot product of 0.9995 or better.</a>
21  public static Quaternion Normalise(this Quaternion q) {
22  float lengthSquared = q.LengthSquared();
23  if ((lengthSquared < 0.02) || (lengthSquared >= 0.8)) return q.normalized; // do it the long way
24 
25  float il = additiveConstant + factor * (lengthSquared - neighborhood);
26  var count = 0;
27 
28  while (((lengthSquared * il * il) < limit) && (count++ < 8)) {
29  il *= additiveConstant + factor * (lengthSquared - neighborhood);
30  }
31 
32  if (count > 4) Debug.LogWarning($"Normalise for {q}, sql {lengthSquared} had {count} iterations");
33 
34  return new Quaternion(q.x * il, q.y * il, q.z * il, q.w * il);
35  }
36 
37  private const float neighborhood = 0.959066f;
38  private const float scale = 1.000311f;
39  private const float limit = 0.9995f * 0.9995f;
40  private static readonly float additiveConstant = scale / Mathf.Sqrt(neighborhood);
41  private static readonly float factor = scale * (-0.5f / (neighborhood * Mathf.Sqrt(neighborhood)));

43  /// <a href="http://bit.ly/2RezNxw">Gyro is right-handed while Unity is left-handed - so change chirality</a>
44  public static Quaternion RightToLeftHanded(this Quaternion q, Trig.Direction axis) {
45  q[axis.Ord] = -q[axis.Ord];
46  q.w = -q.w;
47  return q;
48  }

50  /// <a href="http://bit.ly/2Oq9rtU">Multiplying quaternions may make sense to mathematicians, but to the rest of is RotateBy is clearer</a>

53  /// <a href="http://bit.ly/2Os3WLf">One person's left is another person's up. More importantly the gyroscope in the phone sees forward as the Z axis while Unity likes to use Y for that. This function also changes the chirality (handedness).</a>
54  public static Quaternion SwitchAxis(this Quaternion q, Trig.Direction pivot) {
55  q.Set(-q.x, -q.y, -q.z, -q.w); // change chirality
56  int left = otherAxes[pivot.Ord, 0], right = otherAxes[pivot.Ord, 1];
57  float swap = q[left];
58  q[left] = q[right];
59  q[right] = swap;
60  return q.Normalise();
61  }
