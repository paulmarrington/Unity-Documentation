55  private bool ParseOneEntryToNode() {
56  if (!SkipWhiteSpace()) return false;
57 
58  string key = json[idx++] == '"' ? ParseString() : NextWord(-1);

64  }

69  private void ParseObject() {
70  if (!SkipWhiteSpace()) return;
71 
72  char token = json[idx++];
73 
74  switch (token) {
75  case '{':
76  ParseToNode();
77  return;
78  case '[':
79  ParseArray();
80  return;
81  case '"':
82  tree.Leaf = ParseString();
83  return;

86  break;
87  }

90  }

92  private void ParseArray() {
93  var index = 0;
94 
95  while (SkipWhiteSpace() && (json[idx] != ']')) {
96  using (tree.Anchor()) {
97  tree.Add(index++.ToString());
98  ParseObject();
99  }
100  }
101 
102  idx++; // skip ']'
103  }

105  private string ParseString() {
106  builder.Clear();
107 
108  while ((idx < json.Length) && (json[idx] != '"')) builder.Append(json[idx] == '\\' ? Escape() : json[idx++]);
109 
110  idx++; // drop closing quote
111  SkipWhiteSpace();
112  return builder.ToString();
113  }

115  private char Escape() {
116  switch (json[++idx]) {
117  case 'b': return '\b';
118  case 'f': return '\f';
119  case 'n': return '\n';
120  case 'r': return '\t';
121  case 't': return '\t';
122  default: return json[idx];
123  }
124  }

126  private string NextWord(int offset = 0) {
127  int first = idx += offset;
128  while ((idx < json.Length) && ("{}[]\",:".IndexOf(json[idx]) == -1)) idx++;
129  string word = json.Substring(startIndex: first, length: idx - first);
130  SkipWhiteSpace();
131  return word.Trim();
132  }
133 
134  private static bool IsWhiteSpace(char chr) => char.IsWhiteSpace(chr) || (chr == ',');
135 
136  private bool SkipWhiteSpace() {
137  while ((idx < json.Length) && IsWhiteSpace(json[idx])) idx++;
138  return idx < json.Length;
139  }
