50  private static Contents Fill(string action, string member, string message, string path, int line) {
51  var more = "";
52  int splitIndex = message.IndexOf(" :: ", StringComparison.Ordinal);
53  if (splitIndex != -1) {
54  more = $"{message.Substring(splitIndex + 4)},";
55  message = message.Substring(0, splitIndex);
56  }
57  return new Contents {
58  component = Path.GetFileNameWithoutExtension(path), lineNumber = line, action = action, result = message
59  , extras = $"{more}line={line},member={member}", member = member
60  };
61  }

97  /// <a href="Log Parameter Formatting">Log formatting for extra data</a>
98  public static Dictionary<string, object> ToDictionary(Contents contents) {
99  var dictionary = new Dictionary<string, object> {
100  { "component", contents.component }, { "action", contents.action }, { "result", contents.result }
101  , { "member", contents.member }
102  };
103 
104  for (var i = 0; i < contents.extras.Length; i++) {
105  string key = contents.extras[i].ToString();
106  object value = null;
107 
108  if (key.EndsWith("=")) {
109  key = key.Substring(0, key.Length - 1);
110  value = contents.extras[++i];
111  }
112 
113  dictionary[key] = value;
114  }
115 
116  return dictionary;
117  }
