15  /// <a href="http://bit.ly/2RhLCD5">Find all game objects by name and presence of a type of component</a>
16  public static T[] FindAll<T>(string name) where T : Object {
17  var all = Resources.FindObjectsOfTypeAll<T>();
18  var results = new List<T>();
19 
20  if (string.IsNullOrEmpty(name)) return all;
21 
22  for (var i = 0; i < all.Length; i++) {
23  if (all[i].name == name) results.Add(all[i]);
24  }
25 
26  return results.ToArray();
27  }

29  /// <a href="http://bit.ly/2NUH87i">String representation of the absolute path to a game object</a>
30  public static string Path(GameObject gameObject) {
31  var path = new List<string>();
32  Transform transform = gameObject.transform;
33 
34  while (transform != null) {
35  path.Add(transform.name);
36  transform = transform.parent;
37  }
38 
39  path.Reverse();
40  return string.Join(separator: "/", value: path.ToArray());
41  }

43  /// <a href="http://bit.ly/2NZZYKj">Create a game object on a specific path in the hierarchy</a>
44  public static GameObject CreateGameObject(string path) {
45  var split = path.Split('/');
46  int last = split.Length - 1;
47  var gameObject = GameObject.Find($"/{split[0]}");
48  if (gameObject == null) gameObject = new GameObject(split[0]);
49 
50  for (var i = 1; i <= last; i++) {
51  var transform = gameObject.transform.Find(split[i]);
52 
53  if (transform == null) {
54  transform = new GameObject(split[i]).transform;
55  transform.parent = gameObject.transform;
56  }
57 
58  gameObject = transform.gameObject;
59  }
60 
61  return gameObject;
62  }
