81  private Trees Walk(bool create, string path) {
82  Failed = false;
83  isNumber = -1;
84  var split = path.Split('.');
85 
86  for (var i = 0; i < split.Length; i++) {
87  string key = split[i];
88  if (here == null) return Failure();
89 
90  if (here.Branch[key].Found) {
91  here = here.Branch.Value as Node;
92  } else {
93  if (string.IsNullOrWhiteSpace(key)) {
94  here = i == 0 ? here : here.Parent;
95  } else if (create) {
96  here = (Node) here.Branch.Add(key, Node.New(key, here)).Value;
97  } else {
98  return Failure();
99  }
100  }
101  }
102 
103  return this;
104  }
