20  public T Pick() {
21  if (choices.Length == 0) return default;
22  if (picker != null) return picker();
23 
24  if (choices.Length == 0) { picker = () => default; }
25  else if (!IsRandom) { // cycle through list
26  cycleIndex = 0;
27  picker = () => choices[cycleIndex++ % choices.Length];
28  }
29  else if (choices.Length >= ExhaustiveBelow) { // random selection
30  picker = () => choices[Random.Range(0, choices.Length)];
31  }
32  else {
33  remainingSelections = new List<T>(collection: choices);
34 
35  picker = () => { // different random choice until list exhausted, then repeat
36  if (remainingSelections.Count == 0) remainingSelections = new List<T>(collection: choices);
37 
38  cycleIndex = Random.Range(0, remainingSelections.Count);
39  T result = remainingSelections[index: cycleIndex];
40  remainingSelections.RemoveAt(index: cycleIndex);
41  return result;
42  };
43  }
44 
45  return picker();
46  }
